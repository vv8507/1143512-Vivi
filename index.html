// Sample C++ Code for Student Grade Tracker
#include 
#include 
#include  

using namespace std;


struct Student {
    string name;
    int score;
    char grade;
};


char calculateGrade(int score) {
    if (score >= 90) return 'A';
    else if (score >= 80) return 'B';
    else if (score >= 70) return 'C';
    else if (score >= 60) return 'D';
    else return 'F';
}

int main() {
    const int MAX_STUDENTS = 3;
    
    Student students[MAX_STUDENTS];

    cout << "--- Student Grade Tracker ---" << endl;

    
    for (int i = 0; i < MAX_STUDENTS; ++i) {
        cout << "\nEnter details for Student " << i + 1 << endl;
        
        cout << "Name: ";
        
        cin.ignore(); 
        getline(cin, students[i].name);

        cout << "Score (0-100): ";
        cin >> students[i].score;

        
        students[i].grade = calculateGrade(students[i].score);
    }

    
    cout << "\n======================================" << endl;
    cout << setw(5) << left << "No." 
         << setw(20) << left << "Name" 
         << setw(10) << left << "Score" 
         << "Grade" << endl;
    cout << "======================================" << endl;

    for (int i = 0; i < MAX_STUDENTS; ++i) {
        cout << setw(5) << left << i + 1
             << setw(20) << left << students[i].name
             << setw(10) << left << students[i].score
             << students[i].grade << endl;
    }
    cout << "======================================" << endl;

    return 0;
}

                

// Sample C++ Code for Attendance Checker
#include 
using namespace std;

int main() {
    int total, attended;
    cout << "Total classes: ";
    cin >> total;
    cout << "Classes attended: ";
    cin >> attended;

    double percent = (attended * 100.0) / total;
    cout << "Attendance: " << percent << "%";

    if (percent >= 75) 
        cout << " (Good)";
    else 
        cout << " (Poor)";

    return 0;
}
                

// Sample C++ Banking System
#include 
using namespace std;

class Bank {
private:
    int balance;

public:
    Bank(int b) { balance = b; }

    void deposit(int amt) {
        balance += amt;
    }

    void withdraw(int amt) {
        if (amt > balance)
            cout << "Insufficient funds\n";
        else
            balance -= amt;
    }

    void show() {
        cout << "Balance: " << balance << endl;
    }
};

int main() {
    Bank acc(5000);
    acc.deposit(1000);
    acc.withdraw(2000);
    acc.show();
    return 0;
}
