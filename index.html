// Sample C++ Code for Student Grade Tracker
#include <iostream>
#include <string>
#include <iomanip>

using namespace std;


struct Student {
    string name;
    int score;
    char grade;
};


char calculateGrade(int score) {
    if (score >= 90) return 'A';
    else if (score >= 80) return 'B';
    else if (score >= 70) return 'C';
    else if (score >= 60) return 'D';
    else return 'F';
}

int main() {
    const int MAX_STUDENTS = 3;
    
    Student students[MAX_STUDENTS];

    cout << "--- Student Grade Tracker ---" << endl;

    
    for (int i = 0; i < MAX_STUDENTS; ++i) {
        cout << "\nEnter details for Student " << i + 1 << endl;
        
        cout << "Name: ";
        
        cin.ignore();
        getline(cin, students[i].name);

        cout << "Score (0-100): ";
        cin >> students[i].score;

        
        students[i].grade = calculateGrade(students[i].score);
    }

    
    cout << "\n======================================" << endl;
    cout << setw(5) << left << "No."
             << setw(20) << left << "Name"
             << setw(10) << left << "Score"
             << "Grade" << endl;
    cout << "======================================" << endl;

    for (int i = 0; i < MAX_STUDENTS; ++i) {
        cout << setw(5) << left << i + 1
             << setw(20) << left << students[i].name
             << setw(10) << left << students[i].score
             << students[i].grade << endl;
    }
    cout << "======================================" << endl;

    return 0;
}
